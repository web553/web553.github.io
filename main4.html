33.

int main()
{
    int a[5]={5,1,15,20,25};
    int i,j,m;
    i=++a[1];
    j=a[1]++;
    m=a[i++];
    printf("%d, %d, %d",i,j,m);
    return 0;
}

Output: 3, 2, 15
Explaination:
i = preincrement of a[1] => preincrement of 1 => 2 
j = postincrement of a[1] => postincrement of 1 =>2
m = a[i++] => a[2++] => 15,meanwhile i is incremented to 3
Hence 3, 2, 15
